name: Build EFC Master (PDF + HTML)

on: push: paths: - "docs/efc_master.tex" - "docs/sections/**" - ".github/workflows/build_efc_master.yml" workflow_dispatch:

jobs: build: runs-on: ubuntu-latest

steps: - name: Checkout uses: actions/checkout@v4 with: fetch-depth: 0

- name: Install LaTeX + Pandoc run: | sudo apt-get update sudo apt-get install -y pandoc latexmk  texlive-latex-base texlive-latex-recommended  texlive-latex-extra texlive-fonts-recommended  texlive-science

- name: Build PDF working-directory: docs run: latexmk -pdf -interaction=nonstopmode efc_master.tex

- name: Build HTML (from TeX → MD → HTML) working-directory: docs run: | pandoc efc_master.tex -o efc_master.md

pandoc efc_master.md  –standalone  –include-in-header=mathjax_header.html  –template=efc_master_template.html  –toc  -o efc_master.html

- name: Deploy HTML to docs/ run: | git config user.name "github-actions" git config user.email "actions@github.com"

git add docs/efc_master.html || true git commit -m "Update EFC Master HTML" || true git pull –rebase origin main || true git push origin HEAD:main

- name: Upload PDF uses: actions/upload-artifact@v4 with: name: EFC_Master_PDF path: docs/efc_master.pdf
