% Core functional relations (canonical EFC definitions)

\newcommand{\EfDef}{\Ef = \rhoField \,(1 - \Sfield)}

% Information capacity (C₀-model)

\newcommand{\CcapDef}{\Ccap = C_0 \,(1 - \Sfield)}

% Propagation speed
% c_eff = 1 / (1 + R)
\newcommand{\ceff}{c_{\mathrm{eff}}}
\newcommand{\ceffDef}{\ceff = \frac{1}{1 + \Rfield}}
